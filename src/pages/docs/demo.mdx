```python
import strawberry

@strawberry.type
#^^^^^^^^^^^^^^^ this decorator converts the class to a GraphQL type
class Query:
    hello: str = strawberry.field(resolver=lambda: "Hello world")
    #            ^^^^^^^^^^^^^^^^ this is used to define a resolver

schema = strawberry.Schema(query=Query)
```
